        // Generic page section transition animation helper
        function transitionTo(hideIds, showIds, callback) {
            const hideElems = hideIds.map(id => document.getElementById(id)).filter(Boolean);
            const showElems = showIds.map(id => document.getElementById(id)).filter(Boolean);
            
            // Add page-transition and exit class
            hideElems.forEach(el => {
                el.classList.add('page-transition');
                el.classList.add('page-exit');
            });
            
            // Trigger exit transition
            setTimeout(() => {
                hideElems.forEach(el => {
                    el.classList.add('page-exit-active');
                });
            }, 10);
            
            // Switch sections after exit animation finishes
            setTimeout(() => {
                hideElems.forEach(el => {
                    el.classList.add('hidden-section');
                    el.classList.remove('page-transition', 'page-exit', 'page-exit-active');
                });
                
                if (callback) callback();
                
                showElems.forEach(el => {
                    el.classList.remove('hidden-section');
                    el.classList.add('page-transition', 'page-enter');
                });
                
                // Force layout reflow
                showElems.forEach(el => el.offsetHeight);
                
                // Trigger enter transition
                showElems.forEach(el => {
                    el.classList.add('page-enter-active');
                });
                
                // Clean up transition classes
                setTimeout(() => {
                    showElems.forEach(el => {
                        el.classList.remove('page-transition', 'page-enter', 'page-enter-active');
                    });
                }, 400);
                
            }, 350);
        }

        // --- Dynamic Trending Loading ---
        async function loadTrending() {
            const container = document.getElementById('trending-container');
            if (!container) return;
            container.innerHTML = `<span class="text-xs text-slate-500 animate-pulse">...</span>`;
            try {
                const res = await fetch('/api/trending');
                const data = await res.json();
                if (data.success && Array.isArray(data.trending) && data.trending.length > 0) {
                    container.innerHTML = data.trending.map(title => {
                        const escapedTitle = title.replace(/'/g, "\\'");
                        return `<button onclick="document.getElementById('movie-input').value='${escapedTitle}'; document.getElementById('search-form').dispatchEvent(new Event('submit'))" class="px-3 py-1 rounded-full bg-slate-800/40 hover:bg-slate-700/80 text-slate-300 transition-colors border border-slate-700/50 text-xs">${title}</button>`;
                    }).join('');
                } else {
                    throw new Error("Invalid trending data");
                }
            } catch (e) {
                console.warn("Failed to load trending, using default fallback:", e);
                const fallbacks = currentLang === 'pt'
                    ? ["A Origem", "Breaking Bad", "Interstellar", "A Guerra dos Tronos", "Matrix", "Stranger Things"]
                    : ["Inception", "Breaking Bad", "Interstellar", "Game of Thrones", "The Matrix", "Stranger Things"];
                container.innerHTML = fallbacks.map(title => {
                    const escapedTitle = title.replace(/'/g, "\\'");
                    return `<button onclick="document.getElementById('movie-input').value='${escapedTitle}'; document.getElementById('search-form').dispatchEvent(new Event('submit'))" class="px-3 py-1 rounded-full bg-slate-800/40 hover:bg-slate-700/80 text-slate-300 transition-colors border border-slate-700/50 text-xs">${title}</button>`;
                }).join('');
            }
        }

        async function handleAuthSubmit(e, isLogin) {
            e.preventDefault();
            authError.classList.add('hidden');
            const username = isLogin ? loginUser.value.trim() : registerUser.value.trim();
            const password = isLogin ? loginPass.value : registerPass.value;
            const endpoint = isLogin ? '/api/login' : '/api/register';

            try {
                const res = await fetch(endpoint, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    credentials: 'include',
                    body: JSON.stringify({ username, password })
                });
                const data = await res.json();

                if (res.ok && data.success) {
                    if (isLogin) loginForm.reset(); else registerForm.reset();
                    authModal.classList.remove('active');
                    isLoggedIn = true;
                    await new Promise(r => setTimeout(r, 300));
                    await loadUserWatchlist();
                    setLoginState(true, data.user.username);
                    
                    if (movieInput.value.trim() !== '') {
                        searchForm.dispatchEvent(new Event('submit'));
                    }
                } else {
                    authError.textContent = data.error || dict[currentLang].errAuth;
                    authError.classList.remove('hidden');
                }
            } catch (err) {
                authError.textContent = dict[currentLang].errServer;
                authError.classList.remove('hidden');
            }
        }

        function inWatchlist(title, year) {
            return userWatchlist.some(m => m.title === title && parseInt(m.year) === parseInt(year));
        }

        async function toggleWatchlist(btn, m) {
            const isFav = inWatchlist(m.title, m.year);
            const url = isFav ? '/api/watchlist/remove' : '/api/watchlist/add';
            
            try {
                const res = await fetch(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        imdb_id: m.imdbId || m.imdb_id,
                        title: m.title,
                        original_title: m.originalTitle || m.original_title || m.title,
                        year: m.year,
                        type: m.type || 'movie',
                        rating: m.imdbRating || m.rating,
                        duration: m.duration,
                        category: m.category,
                        description: m.description,
                        trailer_yt_id: m.trailerYtId || m.trailer_yt_id || null
                    })
                });
                
                if (res.status === 401) {
                    openModal('login');
                    return;
                }
                
                const data = await res.json();
                if (data.success) {
                    if (isFav) {
                        userWatchlist = userWatchlist.filter(item => !(item.title === m.title && parseInt(item.year) === parseInt(m.year)));
                    } else {
                        userWatchlist.push(m);
                    }
                    
                    updateWatchlistButton(btn, !isFav);
                    
                    if (activeTab === 'watchlist') {
                        renderWatchlistGrid();
                    }
                }
            } catch(e) {
                console.error("Failed to toggle watchlist:", e);
            }
        }

        // Movie Details Modal Functions
        window.playMovieDirectly = async function(b64, btnElement) {
            if (!b64) return;
            if (btnElement) {
                btnElement.dataset.originalHtml = btnElement.innerHTML;
                btnElement.innerHTML = `<svg class="animate-spin h-3.5 w-3.5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path></svg> <span>Loading...</span>`;
                btnElement.classList.add('opacity-75', 'cursor-not-allowed');
            }
            
            const m = JSON.parse(decodeURIComponent(atob(b64)));
            const title = m.title || '';
            const originalTitle = m.originalTitle || m.original_title || title;
            const year = m.year || '';
            const type = m.type || 'movie';
            let imdbId = m.imdbId || m.imdb_id;
            let tmdbId = m.tmdbId || null;
            
            if (!imdbId) {
                try {
                    const searchRes = await fetch(`https://v3-cinemeta.strem.io/catalog/${type === 'series' || type === 'tv' ? 'series' : 'movie'}/top/search=${encodeURIComponent(originalTitle || title)}.json`);
                    if (searchRes.ok) {
                        const searchData = await searchRes.json();
                        if (searchData.metas && searchData.metas.length > 0) {
                            const match = searchData.metas.find(meta => meta.year == year) || searchData.metas[0];
                            imdbId = match.imdb_id || match.id;
                        }
                    }
                } catch(e) {
                    console.error("Error finding imdbId for play:", e);
                }
            }
            
            if (!tmdbId && imdbId) {
                try {
                    const metaRes = await fetch(`https://v3-cinemeta.strem.io/meta/${type === 'series' || type === 'tv' ? 'series' : 'movie'}/${imdbId}.json`);
                    if (metaRes.ok) {
                        const metaData = await metaRes.json();
                        if (metaData && metaData.meta && metaData.meta.moviedb_id) {
                            tmdbId = metaData.meta.moviedb_id;
                        }
                    }
                } catch(e) {
                    console.error("Error fetching tmdbid:", e);
                }
            }
            
            if (btnElement && btnElement.dataset.originalHtml) {
                btnElement.innerHTML = btnElement.dataset.originalHtml;
                btnElement.classList.remove('opacity-75', 'cursor-not-allowed');
            }
            
            const urlType = (type === 'series' || type === 'tv') ? 'tv' : 'movie';
            const slug = (originalTitle || title).toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-+|-+$/g, '');
            const watchUrl = tmdbId 
                ? `https://flixmomo.app/${urlType}/${tmdbId}/${slug}` 
                : `https://flixmomo.app/search?q=${encodeURIComponent(originalTitle || title)}`;
            
            window.open(watchUrl, '_blank');
        };

        // YouTube Trailer Modal Functions
        function openTrailer(ytId, title = '') {
            const trailerModal = document.getElementById('trailer-modal');
            const container = document.getElementById('trailer-player-container');
            if (!trailerModal || !container || !ytId) return;

            trailerModal.classList.add('active');
            
            // Clean player container and set placeholder loading state
            container.innerHTML = `<div id="yt-player-placeholder" class="w-full h-full rounded-lg bg-slate-950 flex items-center justify-center"><div class="loader"></div></div>`;

            function initPlayer() {
                try {
                    ytPlayer = new YT.Player('yt-player-placeholder', {
                        height: '100%',
                        width: '100%',
                        videoId: ytId,
                        playerVars: {
                            'autoplay': 1,
                            'playsinline': 1,
                            'rel': 0,
                            'modestbranding': 1
                        },
                        events: {
                            'onError': function(e) {
                                console.warn("YouTube Player API Error:", e.data);
                                // Error codes: 2 (invalid parameters), 5 (HTML5 error), 100 (private/removed), 101/150 (restricted/not embeddable)
                                showTrailerFallback(container, title);
                            }
                        }
                    });
                } catch (err) {
                    console.error("Failed to initialize YouTube Player:", err);
                    fallbackIframe(container, ytId);
                }
            }

            if (typeof YT !== 'undefined' && YT.Player) {
                initPlayer();
            } else {
                // If the YouTube API is not loaded yet, poll for up to 3 seconds before falling back
                let attempts = 0;
                const interval = setInterval(() => {
                    attempts++;
                    if (typeof YT !== 'undefined' && YT.Player) {
                        clearInterval(interval);
                        initPlayer();
                    } else if (attempts > 30) {
                        clearInterval(interval);
                        fallbackIframe(container, ytId);
                    }
                }, 100);
            }
        }
        
        function fallbackIframe(container, ytId) {
            container.innerHTML = `<iframe class="w-full h-full rounded-lg" src="https://www.youtube.com/embed/${ytId}?autoplay=1" frameborder="0" allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowfullscreen></iframe>`;
        }

            const trailerModal = document.getElementById('trailer-modal');

        }

        // Social Watchlist Sharing Functions
        async function shareWatchlist(cacheId, btn) {
            const t = dict[currentLang];
            const shareUrl = `${window.location.origin}/?share=${cacheId}`;
            
            let copied = false;
            try {
                if (navigator.clipboard && navigator.clipboard.writeText) {
                    await navigator.clipboard.writeText(shareUrl);
                    copied = true;
                }
            } catch (err) {
                console.warn("navigator.clipboard failed, attempting fallback...", err);
            }

            if (!copied) {
                try {
                    const textArea = document.createElement("textarea");
                    textArea.value = shareUrl;
                    textArea.style.top = "0";
                    textArea.style.left = "0";
                    textArea.style.position = "fixed";
                    textArea.style.opacity = "0";
                    document.body.appendChild(textArea);
                    textArea.focus();
                    textArea.select();
                    const successful = document.execCommand('copy');
                    document.body.removeChild(textArea);
                    if (successful) {
                        copied = true;
                    }
                } catch (err) {
                    console.error("Fallback copy failed:", err);
                }
            }

            if (copied) {
                showToast(t.shareToast);
            } else {
                showToast(currentLang === 'pt' ? "Não foi possível copiar o link automaticamente." : "Could not copy link automatically.");
            }
        }
        
        function showToast(message) {
            let toast = document.getElementById('share-toast');
            if (!toast) {
                toast = document.createElement('div');
                toast.id = 'share-toast';
                toast.className = 'fixed bottom-5 right-5 z-50 glass-panel bg-emerald-950/90 border-emerald-500/40 text-emerald-300 text-xs font-semibold px-4 py-3 rounded-lg shadow-xl pointer-events-none transition-all duration-300 opacity-0 transform translate-y-2';
                document.body.appendChild(toast);
            }
            toast.textContent = message;
            toast.style.opacity = '1';
            toast.style.transform = 'translateY(0)';
            
            setTimeout(() => {
                toast.style.opacity = '0';
                toast.style.transform = 'translateY(8px)';
            }, 3000);
        }
